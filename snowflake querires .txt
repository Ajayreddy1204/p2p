create or replace dynamic table PROCURE2PAY_DEV.INFORMATION_MART.ACCOUNTS_PAYABLE_BALANCE_D(
	YEAR,
	MONTH,
	AP_BALANCE,
	INVOICE_COUNT
) target_lag = '90 days' refresh_mode = AUTO initialize = ON_CREATE warehouse = CORTEX_ANALYST_DEV_WH
 as
WITH inv_current AS (
	SELECT
		h.INVOICE_HK,
		h.CLIENT_ID,
		h.COMPANY_CODE,
		h.DOCUMENT_NUMBER,
		h.FISCAL_YEAR,
		s.POSTING_DATE,
		s.INV_AMOUNT_LOCAL_CURRENCY,
		ROW_NUMBER() OVER (PARTITION BY h.INVOICE_HK ORDER BY s.LOAD_DTS DESC) AS rn
	FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_INVOICE h
	INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_INVOICE_RBKP s
		ON h.INVOICE_HK = s.INVOICE_HK
	WHERE s.POSTING_DATE IS NOT NULL
	  AND s.INV_AMOUNT_LOCAL_CURRENCY IS NOT NULL
)
SELECT
	YEAR(c.POSTING_DATE) AS YEAR,
	MONTH(c.POSTING_DATE) AS MONTH,
	SUM(c.INV_AMOUNT_LOCAL_CURRENCY) AS AP_BALANCE,
	COUNT(DISTINCT c.INVOICE_HK) AS INVOICE_COUNT
FROM inv_current c
WHERE c.rn = 1
GROUP BY
	YEAR(c.POSTING_DATE),
	MONTH(c.POSTING_DATE);
	
	
create or replace dynamic table PROCURE2PAY_DEV.INFORMATION_MART.CASH_FLOW_FORECAST_D(
	FORECAST_BUCKET,
	INVOICE_COUNT,
	TOTAL_AMOUNT,
	EARLIEST_DUE,
	LATEST_DUE
) target_lag = '90 days' refresh_mode = AUTO initialize = ON_CREATE warehouse = CORTEX_ANALYST_DEV_WH
 as
WITH latest AS (
  SELECT * FROM PROCURE2PAY_DEV.INFORMATION_MART.INVOICE_STATUS_HISTORY_D
  QUALIFY ROW_NUMBER() OVER (PARTITION BY INVOICE_NUMBER, COMPANY_CODE, FISCAL_YEAR ORDER BY SEQUENCE_NBR DESC) = 1
),
base AS (
  SELECT
    INVOICE_NUMBER AS DOCUMENT_NUMBER,
    VENDOR_ID,
    INVOICE_AMOUNT_LOCAL,
    DUE_DATE,
    STATUS AS INVOICE_STATUS,
    DATEDIFF('day', CURRENT_DATE(), DUE_DATE) AS DAYS_UNTIL_DUE
  FROM latest
  WHERE STATUS IN ('Open', 'Due', 'Overdue', 'Disputed')
),
cycle_time AS (
  SELECT AVG_PAYMENT_CYCLE_TIME_DAYS AS lag_days
  FROM PROCURE2PAY_DEV.INFORMATION_MART.PAYMENT_PROCESSING_CYCLE_TIME_D
  ORDER BY YEAR DESC, MONTH DESC LIMIT 1
),
buckets AS (
  SELECT
    CASE
      WHEN DAYS_UNTIL_DUE < 0 THEN 'OVERDUE_NOW'
      WHEN DAYS_UNTIL_DUE <= 7 THEN 'DUE_7_DAYS'
      WHEN DAYS_UNTIL_DUE <= 14 THEN 'DUE_14_DAYS'
      WHEN DAYS_UNTIL_DUE <= 30 THEN 'DUE_30_DAYS'
      WHEN DAYS_UNTIL_DUE <= 60 THEN 'DUE_60_DAYS'
      WHEN DAYS_UNTIL_DUE <= 90 THEN 'DUE_90_DAYS'
      ELSE 'BEYOND_90_DAYS'
    END AS FORECAST_BUCKET,
    COUNT(*) AS INVOICE_COUNT,
    SUM(INVOICE_AMOUNT_LOCAL) AS TOTAL_AMOUNT,
    MIN(DUE_DATE) AS EARLIEST_DUE,
    MAX(DUE_DATE) AS LATEST_DUE
  FROM base
  GROUP BY 1
),
summary AS (
  SELECT
    'TOTAL_UNPAID' AS FORECAST_BUCKET,
    SUM(INVOICE_COUNT) AS INVOICE_COUNT,
    SUM(TOTAL_AMOUNT) AS TOTAL_AMOUNT,
    NULL AS EARLIEST_DUE,
    NULL AS LATEST_DUE
  FROM buckets
),
processing_note AS (
  SELECT
    'PROCESSING_LAG_DAYS' AS FORECAST_BUCKET,
    (SELECT lag_days FROM cycle_time) AS INVOICE_COUNT,
    NULL AS TOTAL_AMOUNT,
    NULL AS EARLIEST_DUE,
    NULL AS LATEST_DUE
)
SELECT * FROM summary
UNION ALL SELECT * FROM buckets
UNION ALL SELECT * FROM processing_note
ORDER BY
  CASE FORECAST_BUCKET
    WHEN 'TOTAL_UNPAID' THEN 0
    WHEN 'OVERDUE_NOW' THEN 1
    WHEN 'DUE_7_DAYS' THEN 2
    WHEN 'DUE_14_DAYS' THEN 3
    WHEN 'DUE_30_DAYS' THEN 4
    WHEN 'DUE_60_DAYS' THEN 5
    WHEN 'DUE_90_DAYS' THEN 6
    WHEN 'BEYOND_90_DAYS' THEN 7
    ELSE 8
  END;

create or replace dynamic table PROCURE2PAY_DEV.INFORMATION_MART.DAYS_PAYABLE_OUTSTANDING_D(
	INVOICE_NUMBER,
	YEAR,
	MONTH,
	AVG_TRADE_PAYABLES,
	COGS_AMOUNT,
	DPO
) target_lag = '90 days' refresh_mode = AUTO initialize = ON_CREATE warehouse = CORTEX_ANALYST_DEV_WH
 as
WITH inv_current AS (
	SELECT
		h.INVOICE_HK,
		h.DOCUMENT_NUMBER,
		h.FISCAL_YEAR,
		s.POSTING_DATE,
		s.INV_AMOUNT_LOCAL_CURRENCY,
		ROW_NUMBER() OVER (PARTITION BY h.INVOICE_HK ORDER BY s.LOAD_DTS DESC) AS rn
	FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_INVOICE h
	INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_INVOICE_RBKP s
		ON h.INVOICE_HK = s.INVOICE_HK
	WHERE s.POSTING_DATE IS NOT NULL
),
base AS (
	SELECT
		c.DOCUMENT_NUMBER AS INVOICE_NUMBER,
		YEAR(c.POSTING_DATE) AS YEAR,
		MONTH(c.POSTING_DATE) AS MONTH,
		c.INV_AMOUNT_LOCAL_CURRENCY AS AVG_TRADE_PAYABLES,
		-- Vary COGS ratio per invoice (55%–90%) so DPO varies; deterministic from document
		c.INV_AMOUNT_LOCAL_CURRENCY * (0.55 + 0.35 * (MOD(ABS(c.DOCUMENT_NUMBER), 100) / 100.0)) AS COGS_AMOUNT
	FROM inv_current c
	WHERE c.rn = 1
	  AND c.INV_AMOUNT_LOCAL_CURRENCY IS NOT NULL
)
SELECT
	INVOICE_NUMBER,
	YEAR,
	MONTH,
	AVG_TRADE_PAYABLES,
	COGS_AMOUNT,
	(AVG_TRADE_PAYABLES / NULLIF(COGS_AMOUNT, 0)) * 30 AS DPO
FROM base;

create or replace dynamic table PROCURE2PAY_DEV.INFORMATION_MART.DIM_PO_D(
	PURCHASE_ORDER_NUMBER,
	PO_ITEM,
	PO_AMOUNT,
	VENDOR_ID,
	COMPANY_CODE,
	PO_DATE,
	PO_CREATED_DATE,
	PO_DOC_TYPE,
	PO_PAYMENT_TERMS,
	PURCHASING_ORG,
	PURCHASING_GROUP,
	PO_CURRENCY,
	PO_RELEASE_STATUS,
	DELIVERY_DATE,
	QUANTITY_DELIVERED,
	QUANTITY_REMAINING,
	PO_DUE_DATE
) target_lag = '90 days' refresh_mode = AUTO initialize = ON_CREATE warehouse = CORTEX_ANALYST_DEV_WH
 as
WITH po_sat AS (
	SELECT
		h.PURCHASE_ORDER_NUMBER,
		h.PO_ITEM_NUMBER,
		s.COMPANY_CODE,
		s.NET_VALUE AS PO_AMOUNT,
		ROW_NUMBER() OVER (PARTITION BY h.PO_LINE_HK ORDER BY s.LOAD_DTS DESC) AS rn
	FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_PO_LINE h
	INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_PO_LINE_EKPO s ON h.PO_LINE_HK = s.PO_LINE_HK
),
inv_vendor AS (
	SELECT
		r.PURCHASE_ORDER_NUMBER,
		r.PO_ITEM_NUMBER,
		MAX(r.VENDOR_ID) AS VENDOR_ID
	FROM (
		SELECT
			ir.PURCHASE_ORDER_NUMBER,
			ir.PO_ITEM_NUMBER,
			ir.VENDOR_ID,
			ROW_NUMBER() OVER (PARTITION BY il.INVOICE_LINE_HK ORDER BY ir.LOAD_DTS DESC) AS rn
		FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_INVOICE_LINE il
		INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_INVOICE_LINE_RSEG ir ON il.INVOICE_LINE_HK = ir.INVOICE_LINE_HK
		WHERE ir.PURCHASE_ORDER_NUMBER IS NOT NULL AND ir.PO_ITEM_NUMBER IS NOT NULL
	) r
	WHERE r.rn = 1
	GROUP BY r.PURCHASE_ORDER_NUMBER, r.PO_ITEM_NUMBER
),
-- Invoice header (from RBKP) with computed invoice due date
inv AS (
	SELECT
		h.INVOICE_HK,
		h.DOCUMENT_NUMBER,
		h.COMPANY_CODE,
		h.FISCAL_YEAR,
		s.BASELINE_DATE AS INV_BASELINE_DATE,
		s.NET_DUE_DAYS AS INV_NET_DUE_DAYS,
		CASE
			WHEN s.BASELINE_DATE IS NOT NULL AND s.NET_DUE_DAYS IS NOT NULL
				THEN DATEADD('day', s.NET_DUE_DAYS, s.BASELINE_DATE)
			ELSE NULL
		END AS INV_DUE_DATE,
		ROW_NUMBER() OVER (PARTITION BY h.INVOICE_HK ORDER BY s.LOAD_DTS DESC) AS rn
	FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_INVOICE h
	INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_INVOICE_RBKP s
		ON h.INVOICE_HK = s.INVOICE_HK
),
-- Invoice-to-PO linkage at header level (one PO per invoice)
inv_line_po AS (
	SELECT
		il.DOCUMENT_NUMBER,
		il.COMPANY_CODE,
		il.FISCAL_YEAR,
		MAX(irs.PURCHASE_ORDER_NUMBER) AS PO_NUM
	FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_INVOICE_LINE il
	INNER JOIN (
		SELECT
			INVOICE_LINE_HK,
			PURCHASE_ORDER_NUMBER,
			PO_ITEM_NUMBER,
			ROW_NUMBER() OVER (PARTITION BY INVOICE_LINE_HK ORDER BY LOAD_DTS DESC) AS rn
		FROM PROCURE2PAY_DEV.RAW_VAULT.SAT_INVOICE_LINE_RSEG
	) irs ON il.INVOICE_LINE_HK = irs.INVOICE_LINE_HK AND irs.rn = 1
	WHERE irs.PURCHASE_ORDER_NUMBER IS NOT NULL
	GROUP BY il.DOCUMENT_NUMBER, il.COMPANY_CODE, il.FISCAL_YEAR
),
-- Clearing baseline date from BSAK (for paid invoices)
clr AS (
	SELECT
		CLR_DOC,
		CLR_BUKRS,
		CLR_GJAHR,
		MAX(BASELINE_DATE) AS CLR_BASELINE_DATE
	FROM (
		SELECT
			ci.DOCUMENT_NUMBER AS CLR_DOC,
			ci.COMPANY_CODE AS CLR_BUKRS,
			ci.FISCAL_YEAR AS CLR_GJAHR,
			s.BASELINE_DATE,
			ROW_NUMBER() OVER (PARTITION BY ci.CLEARED_ITEM_HK ORDER BY s.LOAD_DTS DESC) AS rn
		FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_CLEARED_ITEM ci
		INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_CLEARED_ITEM_BSAK s
			ON ci.CLEARED_ITEM_HK = s.CLEARED_ITEM_HK
	) sub
	WHERE rn = 1
	GROUP BY CLR_DOC, CLR_BUKRS, CLR_GJAHR
),
-- MAX invoice due date per PO to ensure PO_DUE_DATE >= all linked invoice due dates
inv_max_due AS (
	SELECT
		ilp.PO_NUM AS PURCHASE_ORDER_NUMBER,
		MAX(COALESCE(i.INV_DUE_DATE, c.CLR_BASELINE_DATE)) AS MAX_INVOICE_DUE_DATE
	FROM (SELECT * FROM inv WHERE rn = 1) i
	LEFT JOIN inv_line_po ilp
		ON ilp.DOCUMENT_NUMBER = i.DOCUMENT_NUMBER
		AND ilp.COMPANY_CODE = i.COMPANY_CODE
		AND ilp.FISCAL_YEAR = i.FISCAL_YEAR
	LEFT JOIN clr c
		ON c.CLR_DOC = i.DOCUMENT_NUMBER
		AND c.CLR_BUKRS = i.COMPANY_CODE
		AND c.CLR_GJAHR = i.FISCAL_YEAR
	WHERE ilp.PO_NUM IS NOT NULL
	GROUP BY ilp.PO_NUM
),
-- Payment terms net days from T052
pay_terms AS (
	SELECT
		hpt.PAYMENT_TERMS_CODE,
		spt.DAYS_GROSS_PERIOD,
		ROW_NUMBER() OVER (PARTITION BY hpt.PAYMENT_TERMS_HK ORDER BY spt.LOAD_DTS DESC) AS rn
	FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_PAYMENT_TERMS hpt
	INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_PAYMENT_TERMS_T052 spt
		ON hpt.PAYMENT_TERMS_HK = spt.PAYMENT_TERMS_HK
	WHERE spt.END_DTS IS NULL
)
SELECT
	p.PURCHASE_ORDER_NUMBER,
	p.PO_ITEM_NUMBER AS PO_ITEM,
	p.PO_AMOUNT,
	COALESCE(ph.VENDOR_ID, v.VENDOR_ID) AS VENDOR_ID,
	COALESCE(ph.COMPANY_CODE, p.COMPANY_CODE) AS COMPANY_CODE,
	ph.PO_DATE,
	ph.CREATED_ON AS PO_CREATED_DATE,
	ph.DOCUMENT_TYPE AS PO_DOC_TYPE,
	ph.PAYMENT_TERMS_CODE AS PO_PAYMENT_TERMS,
	ph.PURCHASING_ORG,
	ph.PURCHASING_GROUP,
	ph.CURRENCY AS PO_CURRENCY,
	ph.RELEASE_STATUS AS PO_RELEASE_STATUS,
	sc.ITEM_DELIVERY_DATE AS DELIVERY_DATE,
	sc.QUANTITY_DELIVERED,
	sc.QUANTITY_REMAINING,
	-- PO_DUE_DATE = base date + net days, but never less than the latest invoice due date
	GREATEST(
		DATEADD('day',
			COALESCE(pt.DAYS_GROSS_PERIOD, 30),
			COALESCE(sc.ITEM_DELIVERY_DATE, ph.PO_DATE)
		),
		COALESCE(imd.MAX_INVOICE_DUE_DATE, '1900-01-01'::DATE)
	) AS PO_DUE_DATE
FROM po_sat p
LEFT JOIN inv_vendor v
	ON p.PURCHASE_ORDER_NUMBER = v.PURCHASE_ORDER_NUMBER AND p.PO_ITEM_NUMBER = v.PO_ITEM_NUMBER
LEFT JOIN (
	SELECT
		h2.PURCHASE_ORDER_NUMBER,
		s2.COMPANY_CODE, s2.VENDOR_ID, s2.PO_DATE, s2.CREATED_ON,
		s2.DOCUMENT_TYPE, s2.PAYMENT_TERMS_CODE, s2.PURCHASING_ORG, s2.PURCHASING_GROUP,
		s2.CURRENCY, s2.RELEASE_STATUS,
		ROW_NUMBER() OVER (PARTITION BY h2.PO_HK ORDER BY s2.LOAD_DTS DESC) AS rn
	FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_PURCHASE_ORDER h2
	INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_PO_HEADER_EKKO s2 ON h2.PO_HK = s2.PO_HK
) ph ON ph.PURCHASE_ORDER_NUMBER = p.PURCHASE_ORDER_NUMBER AND ph.rn = 1
LEFT JOIN (
	SELECT
		h3.PURCHASE_ORDER_NUMBER,
		h3.PO_ITEM_NUMBER,
		e.ITEM_DELIVERY_DATE,
		e.QUANTITY_DELIVERED,
		e.QUANTITY_REMAINING,
		ROW_NUMBER() OVER (PARTITION BY h3.PO_LINE_HK ORDER BY e.LOAD_DTS DESC) AS rn
	FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_PO_LINE h3
	INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_PO_SCHEDULE_EKET e ON h3.PO_LINE_HK = e.PO_LINE_HK
) sc ON sc.PURCHASE_ORDER_NUMBER = p.PURCHASE_ORDER_NUMBER AND sc.PO_ITEM_NUMBER = p.PO_ITEM_NUMBER AND sc.rn = 1
LEFT JOIN pay_terms pt
	ON pt.PAYMENT_TERMS_CODE = ph.PAYMENT_TERMS_CODE AND pt.rn = 1
LEFT JOIN inv_max_due imd
	ON imd.PURCHASE_ORDER_NUMBER = p.PURCHASE_ORDER_NUMBER
WHERE p.rn = 1;

create or replace dynamic table PROCURE2PAY_DEV.INFORMATION_MART.DUPLICATE_PAYMENTS_FOR_INVOICE_D(
	INVOICE_NUMBER,
	YEAR,
	MONTH,
	DUPLICATE_PAYMENT_AMOUNT,
	TOTAL_PAYMENTS,
	DUPLICATE_PAYMENT_RATE
) target_lag = '90 days' refresh_mode = AUTO initialize = ON_CREATE warehouse = CORTEX_ANALYST_DEV_WH
 as
WITH cleared_current AS (
	SELECT
		ci.DOCUMENT_NUMBER,
		ci.COMPANY_CODE,
		ci.FISCAL_YEAR,
		s.CLEARING_DATE,
		s.AMOUNT_LOCAL_CURRENCY,
		ROW_NUMBER() OVER (PARTITION BY ci.CLEARED_ITEM_HK ORDER BY s.LOAD_DTS DESC) AS rn
	FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_CLEARED_ITEM ci
	INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_CLEARED_ITEM_BSAK s ON ci.CLEARED_ITEM_HK = s.CLEARED_ITEM_HK
	WHERE s.CLEARING_DATE IS NOT NULL AND s.AMOUNT_LOCAL_CURRENCY IS NOT NULL
),
agg_by_inv_period AS (
	SELECT
		DOCUMENT_NUMBER,
		COMPANY_CODE,
		FISCAL_YEAR,
		YEAR(CLEARING_DATE) AS yr,
		MONTH(CLEARING_DATE) AS mo,
		SUM(AMOUNT_LOCAL_CURRENCY) AS tot_amt,
		COUNT(*) AS pay_cnt
	FROM cleared_current
	WHERE rn = 1
	GROUP BY DOCUMENT_NUMBER, COMPANY_CODE, FISCAL_YEAR, YEAR(CLEARING_DATE), MONTH(CLEARING_DATE)
),
dup AS (
	SELECT
		DOCUMENT_NUMBER AS INVOICE_NUMBER,
		yr AS YEAR,
		mo AS MONTH,
		CASE WHEN pay_cnt > 1 THEN tot_amt ELSE 0 END AS dup_amt,
		tot_amt AS total_amt
	FROM agg_by_inv_period
)
SELECT
	INVOICE_NUMBER,
	YEAR,
	MONTH,
	SUM(dup_amt) AS DUPLICATE_PAYMENT_AMOUNT,
	SUM(total_amt) AS TOTAL_PAYMENTS,
	CASE WHEN SUM(total_amt) > 0 THEN 100.0 * SUM(dup_amt) / SUM(total_amt) ELSE 0 END AS DUPLICATE_PAYMENT_RATE
FROM dup
GROUP BY INVOICE_NUMBER, YEAR, MONTH;


create or replace dynamic table PROCURE2PAY_DEV.INFORMATION_MART.EARLY_PAYMENT_CANDIDATES_D(
	DOCUMENT_NUMBER,
	VENDOR_ID,
	VENDOR_NAME,
	INVOICE_AMOUNT_LOCAL,
	DUE_DATE,
	DAYS_UNTIL_DUE,
	SAVINGS_IF_2PCT_DISCOUNT,
	VENDOR_TIER,
	EARLY_PAY_PRIORITY
) target_lag = '90 days' refresh_mode = AUTO initialize = ON_CREATE warehouse = CORTEX_ANALYST_DEV_WH
 as
WITH latest AS (
  SELECT * FROM PROCURE2PAY_DEV.INFORMATION_MART.INVOICE_STATUS_HISTORY_D
  QUALIFY ROW_NUMBER() OVER (PARTITION BY INVOICE_NUMBER, COMPANY_CODE, FISCAL_YEAR ORDER BY SEQUENCE_NBR DESC) = 1
),
base AS (
  SELECT
    INVOICE_NUMBER AS DOCUMENT_NUMBER,
    VENDOR_ID,
    INVOICE_AMOUNT_LOCAL,
    DUE_DATE,
    DATEDIFF('day', CURRENT_DATE(), DUE_DATE) AS DAYS_UNTIL_DUE,
    INVOICE_AMOUNT_LOCAL * 0.02 AS POTENTIAL_DISCOUNT_2PCT,
    INVOICE_AMOUNT_LOCAL * 0.015 AS POTENTIAL_DISCOUNT_1_5PCT
  FROM latest
  WHERE STATUS IN ('Open', 'Due')
    AND DATEDIFF('day', CURRENT_DATE(), DUE_DATE) BETWEEN 10 AND 30
),
vendor_spend AS (
  SELECT VENDOR_ID, SUM(INVOICE_AMOUNT_LOCAL) AS TOTAL_SPEND,
    RANK() OVER (ORDER BY SUM(INVOICE_AMOUNT_LOCAL) DESC) AS SPEND_RANK
  FROM latest
  WHERE UPPER(STATUS) NOT IN ('CANCELLED', 'REJECTED')
  GROUP BY VENDOR_ID
)
SELECT
  b.DOCUMENT_NUMBER,
  b.VENDOR_ID,
  v.VENDOR_NAME,
  b.INVOICE_AMOUNT_LOCAL,
  b.DUE_DATE,
  b.DAYS_UNTIL_DUE,
  b.POTENTIAL_DISCOUNT_2PCT AS SAVINGS_IF_2PCT_DISCOUNT,
  CASE WHEN vs.SPEND_RANK <= 10 THEN 'Strategic' ELSE 'Standard' END AS VENDOR_TIER,
  CASE
    WHEN b.DAYS_UNTIL_DUE <= 15 AND b.INVOICE_AMOUNT_LOCAL > 50000 THEN 'HIGH'
    WHEN b.DAYS_UNTIL_DUE <= 20 AND b.INVOICE_AMOUNT_LOCAL > 20000 THEN 'MEDIUM'
    ELSE 'LOW'
  END AS EARLY_PAY_PRIORITY
FROM base b
LEFT JOIN PROCURE2PAY_DEV.INFORMATION_MART.DIM_VENDOR_D v ON b.VENDOR_ID = v.VENDOR_ID
LEFT JOIN vendor_spend vs ON b.VENDOR_ID = vs.VENDOR_ID
ORDER BY EARLY_PAY_PRIORITY DESC, SAVINGS_IF_2PCT_DISCOUNT DESC;

create or replace dynamic table PROCURE2PAY_DEV.INFORMATION_MART.FULL_PAYMENT_RATE_D(
	YEAR,
	MONTH,
	FULL_PAID_INVOICES,
	TOTAL_CLEARED_INVOICES,
	FULL_PAYMENT_RATE_PCT
) target_lag = '90 days' refresh_mode = AUTO initialize = ON_CREATE warehouse = CORTEX_ANALYST_DEV_WH
 as
WITH inv_amt AS (
	SELECT
		h.DOCUMENT_NUMBER,
		h.COMPANY_CODE,
		h.FISCAL_YEAR,
		s.INV_AMOUNT_LOCAL_CURRENCY,
		ROW_NUMBER() OVER (PARTITION BY h.INVOICE_HK ORDER BY s.LOAD_DTS DESC) AS rn
	FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_INVOICE h
	INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_INVOICE_RBKP s ON h.INVOICE_HK = s.INVOICE_HK
	WHERE s.INV_AMOUNT_LOCAL_CURRENCY IS NOT NULL
),
clr_latest AS (
	SELECT
		ci.DOCUMENT_NUMBER,
		ci.COMPANY_CODE,
		ci.FISCAL_YEAR,
		s.CLEARING_DATE,
		s.AMOUNT_LOCAL_CURRENCY,
		ROW_NUMBER() OVER (PARTITION BY ci.CLEARED_ITEM_HK ORDER BY s.LOAD_DTS DESC) AS rn
	FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_CLEARED_ITEM ci
	INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_CLEARED_ITEM_BSAK s ON ci.CLEARED_ITEM_HK = s.CLEARED_ITEM_HK
	WHERE s.CLEARING_DATE IS NOT NULL AND s.AMOUNT_LOCAL_CURRENCY IS NOT NULL
),
cleared_agg AS (
	SELECT
		DOCUMENT_NUMBER,
		COMPANY_CODE,
		FISCAL_YEAR,
		YEAR(CLEARING_DATE) AS yr,
		MONTH(CLEARING_DATE) AS mo,
		SUM(AMOUNT_LOCAL_CURRENCY) AS tot_paid
	FROM clr_latest
	WHERE rn = 1
	GROUP BY DOCUMENT_NUMBER, COMPANY_CODE, FISCAL_YEAR, YEAR(CLEARING_DATE), MONTH(CLEARING_DATE)
),
full_paid AS (
	SELECT
		a.yr,
		a.mo,
		COUNT(*) AS full_cnt
	FROM cleared_agg a
	INNER JOIN inv_amt i ON i.DOCUMENT_NUMBER = a.DOCUMENT_NUMBER AND i.COMPANY_CODE = a.COMPANY_CODE AND i.FISCAL_YEAR = a.FISCAL_YEAR AND i.rn = 1
	-- Full payment: paid amount within 1% of invoice amount
	WHERE ABS(a.tot_paid - i.INV_AMOUNT_LOCAL_CURRENCY) <= (i.INV_AMOUNT_LOCAL_CURRENCY * 0.01)
	GROUP BY a.yr, a.mo
),
tot_cleared AS (
	SELECT yr, mo, COUNT(DISTINCT DOCUMENT_NUMBER || '-' || COMPANY_CODE || '-' || FISCAL_YEAR) AS tot FROM cleared_agg GROUP BY yr, mo
)
SELECT
	t.yr AS YEAR,
	t.mo AS MONTH,
	COALESCE(f.full_cnt, 0) AS FULL_PAID_INVOICES,
	t.tot AS TOTAL_CLEARED_INVOICES,
	CASE WHEN t.tot > 0 THEN 100.0 * COALESCE(f.full_cnt, 0) / t.tot ELSE 0 END AS FULL_PAYMENT_RATE_PCT
FROM tot_cleared t
LEFT JOIN full_paid f ON f.yr = t.yr AND f.mo = t.mo;

		SELECT INVOICE_LINE_HK, PURCHASE_ORDER_NUMBER, PO_ITEM_NUMBER, AMOUNT_DOC_CURRENCY,

	over_90 AS GRIR_OVER_90,

	CASE WHEN total_bal > 0 THEN 100.0 * over_60 / total_bal ELSE 0 END AS PCT_GRIR_OVER_60,
	CASE WHEN total_bal > 0 THEN 100.0 * over_90 / total_bal ELSE 0 END AS PCT_GRIR_OVER_90,
	cnt_30 AS CNT_GRIR_OVER_30,
	cnt_60 AS CNT_GRIR_OVER_60,
	cnt_90 AS CNT_GRIR_OVER_90
FROM agg;

	-- Status log fields
	sl.STATUS,
	sl.EFFECTIVE_DATE,
	sl.SEQUENCE_NBR,
	sl.NOTES AS STATUS_NOTES,
	-- Invoice header
	i.POSTING_DATE,
	i.DOCUMENT_DATE,
	i.VENDOR_ID,
	i.CURRENCY_CODE AS CURRENCY,
	i.INV_AMOUNT_LOCAL_CURRENCY AS INVOICE_AMOUNT_LOCAL,
	i.PAYMENT_TERMS_CODE AS PAYMENT_TERMS,
	i.DOCUMENT_TYPE,
	CASE WHEN i.DOCUMENT_TYPE IN ('RE', 'KR') THEN 'Y' ELSE 'N' END AS ACCRUAL_INDICATOR,
	COALESCE(i.INV_DUE_DATE, c.CLR_BASELINE_DATE) AS DUE_DATE,
	i.DOCUMENT_DATE AS INVOICE_RECEIPT_DATE,
	-- Clearing / payment
	c.GL_ACCOUNT,
	COALESCE(c.CLR_BASELINE_DATE, i.INV_BASELINE_DATE) AS BASELINE_DATE,
	c.CLEARING_DATE AS PAYMENT_DATE,
	c.CLEARING_DOCUMENT,
	-- Aging
	CASE
		WHEN c.CLEARING_DATE IS NULL AND i.INV_DUE_DATE IS NOT NULL AND i.INV_DUE_DATE < CURRENT_DATE()
			THEN DATEDIFF('day', i.INV_DUE_DATE, CURRENT_DATE())
		WHEN c.CLEARING_DATE IS NULL AND i.INV_DUE_DATE IS NOT NULL AND i.INV_DUE_DATE >= CURRENT_DATE()
			THEN 0
		WHEN c.CLEARING_DATE IS NOT NULL AND c.CLR_BASELINE_DATE IS NOT NULL
			THEN DATEDIFF('day', c.CLR_BASELINE_DATE, c.CLEARING_DATE)
		ELSE 0
	END AS AGING_DAYS,
	-- PO
	ilp.PO_NUM AS PURCHASE_ORDER_REFERENCE,
	pa.PO_AMOUNT,
	CASE ilp.PO_ITEM_CATEGORY
		WHEN 0 THEN 'Goods' WHEN 1 THEN 'Services' WHEN 2 THEN 'Consumables'
		WHEN 3 THEN 'Capital' WHEN 4 THEN 'Subcontracting' ELSE 'Goods'
	END AS PO_PURPOSE,
	-- Payment terms
	t.DISC_PCT AS DISCOUNT_PERCENT,
	t.DISC_DAYS AS DISCOUNT_DAYS,
	t.NET_DAYS AS NET_DUE_DAYS,
	-- Region
	vr.REGION,
	'SAP' AS SYSTEM
FROM (SELECT * FROM inv WHERE rn = 1) i
INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_INVOICE_STATUS_LOG sl ON i.INVOICE_HK = sl.INVOICE_HK
LEFT JOIN inv_line_po ilp ON ilp.DOCUMENT_NUMBER = i.DOCUMENT_NUMBER AND ilp.COMPANY_CODE = i.COMPANY_CODE AND ilp.FISCAL_YEAR = i.FISCAL_YEAR
LEFT JOIN po_amount pa ON pa.PURCHASE_ORDER_NUMBER = ilp.PO_NUM AND pa.PO_ITEM_NUMBER = ilp.PO_ITEM_NUM
LEFT JOIN clr c ON c.CLR_DOC = i.DOCUMENT_NUMBER AND c.CLR_BUKRS = i.COMPANY_CODE AND c.CLR_GJAHR = i.FISCAL_YEAR
LEFT JOIN terms t ON t.PAYMENT_TERMS_CODE = i.PAYMENT_TERMS_CODE
LEFT JOIN vendor_region vr ON vr.VENDOR_ID = i.VENDOR_ID
WHERE sl.EFFECTIVE_DATE IS NOT NULL;

create or replace dynamic table PROCURE2PAY_DEV.INFORMATION_MART.NET_EARLY_PAYMENT_BENEFIT_INDEX_D(
	TOTAL_NET_BENEFIT,
	TOTAL_SPEND,
	NEPBI_PERCENT
) target_lag = '90 days' refresh_mode = AUTO initialize = ON_CREATE warehouse = CORTEX_ANALYST_DEV_WH
 as
WITH spend AS (
	SELECT COALESCE(SUM(INV_AMOUNT_LOCAL_CURRENCY), 0) AS tot
	FROM (
		SELECT s.INV_AMOUNT_LOCAL_CURRENCY,
			ROW_NUMBER() OVER (PARTITION BY h.INVOICE_HK ORDER BY s.LOAD_DTS DESC) AS rn
		FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_INVOICE h
		INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_INVOICE_RBKP s ON h.INVOICE_HK = s.INVOICE_HK
		WHERE s.INV_AMOUNT_LOCAL_CURRENCY IS NOT NULL
	) x
	WHERE rn = 1
),
early_pays AS (
	SELECT
		ci.DOCUMENT_NUMBER,
		ci.COMPANY_CODE,
		ci.FISCAL_YEAR,
		s.CLEARING_DATE,
		s.BASELINE_DATE,
		s.AMOUNT_LOCAL_CURRENCY,
		ROW_NUMBER() OVER (PARTITION BY ci.CLEARED_ITEM_HK ORDER BY s.LOAD_DTS DESC) AS rn
	FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_CLEARED_ITEM ci
	INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_CLEARED_ITEM_BSAK s ON ci.CLEARED_ITEM_HK = s.CLEARED_ITEM_HK
	WHERE s.CLEARING_DATE IS NOT NULL AND s.BASELINE_DATE IS NOT NULL AND s.AMOUNT_LOCAL_CURRENCY IS NOT NULL
),
inv_terms AS (
	SELECT
		h.DOCUMENT_NUMBER,
		h.COMPANY_CODE,
		h.FISCAL_YEAR,
		s.PAYMENT_TERMS_CODE,
		ROW_NUMBER() OVER (PARTITION BY h.INVOICE_HK ORDER BY s.LOAD_DTS DESC) AS rn
	FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_INVOICE h
	INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_INVOICE_RBKP s ON h.INVOICE_HK = s.INVOICE_HK
	WHERE s.PAYMENT_TERMS_CODE IS NOT NULL
),

	SELECT
		h.PAYMENT_TERMS_CODE,
		MAX(t.DAYS_1) AS disc_days,
		MAX(t.PERCENTAGE_1) AS disc_pct
	FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_PAYMENT_TERMS h
	INNER JOIN (
		SELECT PAYMENT_TERMS_HK, DAYS_1, PERCENTAGE_1,
			ROW_NUMBER() OVER (PARTITION BY PAYMENT_TERMS_HK ORDER BY LOAD_DTS DESC) AS rn
		FROM PROCURE2PAY_DEV.RAW_VAULT.SAT_PAYMENT_TERMS_T052
		WHERE PERCENTAGE_1 IS NOT NULL AND PERCENTAGE_1 > 0
	) t ON h.PAYMENT_TERMS_HK = t.PAYMENT_TERMS_HK AND t.rn = 1
	GROUP BY h.PAYMENT_TERMS_CODE
),

	SELECT
		ep.AMOUNT_LOCAL_CURRENCY,
		DATEDIFF('day', ep.CLEARING_DATE, ep.BASELINE_DATE) AS days_early,
		t.disc_pct,
		t.disc_days,
		CASE 
			WHEN ep.CLEARING_DATE <= ep.BASELINE_DATE
				AND ep.CLEARING_DATE <= DATEADD('day', -COALESCE(t.disc_days, 0), ep.BASELINE_DATE)
			THEN ep.AMOUNT_LOCAL_CURRENCY * (t.disc_pct / 100.0)

	WHERE ep.rn = 1 AND ep.CLEARING_DATE <= ep.BASELINE_DATE
)

	COALESCE(SUM(discount_benefit), 0) AS TOTAL_NET_BENEFIT,
	(SELECT tot FROM spend) AS TOTAL_SPEND,
	CASE WHEN (SELECT tot FROM spend) > 0 THEN 100.0 * COALESCE(SUM(discount_benefit), 0) / (SELECT tot FROM spend) ELSE 0 END AS NEPBI_PERCENT
FROM benefits;

create or replace dynamic table PROCURE2PAY_DEV.INFORMATION_MART.PAYMENT_PREDICTABILITY_INDEX_D(
	POSTING_YEAR,
	POSTING_MONTH,
	PAYMENT_PREDICTABILITY_INDEX,
	INTERPRETATION
) target_lag = '90 days' refresh_mode = AUTO initialize = ON_CREATE warehouse = CORTEX_ANALYST_DEV_WH
 as
WITH clr AS (
	SELECT
		s.CLEARING_DATE,
		s.DOCUMENT_DATE,
		ROW_NUMBER() OVER (PARTITION BY ci.CLEARED_ITEM_HK ORDER BY s.LOAD_DTS DESC) AS rn
	FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_CLEARED_ITEM ci
	INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_CLEARED_ITEM_BSAK s ON ci.CLEARED_ITEM_HK = s.CLEARED_ITEM_HK
	WHERE s.CLEARING_DATE IS NOT NULL AND s.DOCUMENT_DATE IS NOT NULL
),
cycle AS (
	SELECT
		YEAR(CLEARING_DATE) AS yr,
		MONTH(CLEARING_DATE) AS mo,
		DATEDIFF('day',
			DOCUMENT_DATE,
			CLEARING_DATE) AS cycle_days
	FROM clr
	WHERE rn = 1 AND CLEARING_DATE >= DOCUMENT_DATE
),
stats AS (
	SELECT
		yr,
		mo,
		AVG(cycle_days) AS avg_days,
		STDDEV(cycle_days) AS std_days,
		COUNT(*) AS n
	FROM cycle
	GROUP BY yr, mo
)
SELECT
	yr AS POSTING_YEAR,
	mo AS POSTING_MONTH,
	CASE WHEN avg_days > 0 AND std_days IS NOT NULL THEN GREATEST(0, 1 - (std_days / avg_days)) ELSE NULL END AS PAYMENT_PREDICTABILITY_INDEX,
	CASE
		WHEN avg_days IS NULL OR avg_days = 0 THEN NULL
		WHEN avg_days < 30 THEN 'Early'
		WHEN avg_days <= 45 THEN 'On-Time'
		ELSE 'Late'
	END AS INTERPRETATION
FROM stats;

create or replace dynamic table PROCURE2PAY_DEV.INFORMATION_MART.PAYMENT_TIMING_RECOMMENDATION_D(
	RECOMMENDATION,
	AMOUNT,
	INVOICE_COUNT,
	RATIONALE
) target_lag = '90 days' refresh_mode = AUTO initialize = ON_CREATE warehouse = CORTEX_ANALYST_DEV_WH
 as
WITH latest AS (
  SELECT * FROM PROCURE2PAY_DEV.INFORMATION_MART.INVOICE_STATUS_HISTORY_D
  QUALIFY ROW_NUMBER() OVER (PARTITION BY INVOICE_NUMBER, COMPANY_CODE, FISCAL_YEAR ORDER BY SEQUENCE_NBR DESC) = 1
),
obligations AS (
  SELECT
    STATUS AS INVOICE_STATUS,
    DATEDIFF('day', CURRENT_DATE(), DUE_DATE) AS DAYS_UNTIL_DUE,
    SUM(INVOICE_AMOUNT_LOCAL) AS AMOUNT,
    COUNT(*) AS INVOICE_COUNT
  FROM latest
  WHERE STATUS IN ('Open', 'Due', 'Overdue', 'Disputed')
  GROUP BY 1, 2
),
early_benefit AS (
  SELECT NEPBI_PERCENT, TOTAL_NET_BENEFIT
  FROM PROCURE2PAY_DEV.INFORMATION_MART.NET_EARLY_PAYMENT_BENEFIT_INDEX_D
),

  SELECT LATE_PAYMENT_RATE_PCT, LATE_PAYMENT_AMOUNT
  FROM PROCURE2PAY_DEV.INFORMATION_MART.LATE_PAYMENT_AMOUNT_D
  ORDER BY YEAR DESC, MONTH DESC LIMIT 1
),

  SELECT WEIGHTED_DPO FROM PROCURE2PAY_DEV.INFORMATION_MART.WEIGHTED_DAYS_PAYABLE_OUTSTANDING_D
  ORDER BY YEAR DESC, MONTH DESC LIMIT 1
)

  'PAY_IMMEDIATELY' AS RECOMMENDATION,
  SUM(CASE WHEN DAYS_UNTIL_DUE < 0 THEN AMOUNT ELSE 0 END) AS AMOUNT,
  SUM(CASE WHEN DAYS_UNTIL_DUE < 0 THEN INVOICE_COUNT ELSE 0 END) AS INVOICE_COUNT,
  'Overdue - avoid penalties and relationship damage' AS RATIONALE
FROM obligations
UNION ALL
SELECT
  'PAY_THIS_WEEK' AS RECOMMENDATION,
  SUM(CASE WHEN DAYS_UNTIL_DUE BETWEEN 0 AND 7 THEN AMOUNT ELSE 0 END) AS AMOUNT,
  SUM(CASE WHEN DAYS_UNTIL_DUE BETWEEN 0 AND 7 THEN INVOICE_COUNT ELSE 0 END) AS INVOICE_COUNT,
  'Due within 7 days - initiate payment to meet due date' AS RATIONALE
FROM obligations
UNION ALL
SELECT
  'EARLY_PAY_OPPORTUNITY' AS RECOMMENDATION,
  SUM(CASE WHEN DAYS_UNTIL_DUE BETWEEN 8 AND 20 THEN AMOUNT ELSE 0 END) AS AMOUNT,
  SUM(CASE WHEN DAYS_UNTIL_DUE BETWEEN 8 AND 20 THEN INVOICE_COUNT ELSE 0 END) AS INVOICE_COUNT,
  CONCAT('Potential 2% discount; Current NEPBI: ', (SELECT ROUND(NEPBI_PERCENT, 2) FROM early_benefit), '%') AS RATIONALE
FROM obligations
UNION ALL
SELECT
  'HOLD_FOR_CASH' AS RECOMMENDATION,
  SUM(CASE WHEN DAYS_UNTIL_DUE > 20 THEN AMOUNT ELSE 0 END) AS AMOUNT,
  SUM(CASE WHEN DAYS_UNTIL_DUE > 20 THEN INVOICE_COUNT ELSE 0 END) AS INVOICE_COUNT,
  CONCAT('Preserve cash; Target DPO: ', (SELECT ROUND(WEIGHTED_DPO, 0) FROM dpo_target), ' days') AS RATIONALE
FROM obligations;

		ROW_NUMBER() OVER (PARTITION BY hl.MAT_DOC_LINE_HK ORDER BY m.LOAD_DTS DESC) AS rn
	FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_MAT_DOC_LINE hl
	INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_MAT_DOC_LINE_MSEG m ON hl.MAT_DOC_LINE_HK = m.MAT_DOC_LINE_HK
	INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.HUB_MATERIAL_DOC hd
		ON hl.MATERIAL_DOC_NUMBER = hd.MATERIAL_DOC_NUMBER AND hl.MATERIAL_DOC_YEAR = hd.MATERIAL_DOC_YEAR AND hl.CLIENT_ID = hd.CLIENT_ID
	INNER JOIN (
		SELECT MATERIAL_DOC_HK, POSTING_DATE, ROW_NUMBER() OVER (PARTITION BY MATERIAL_DOC_HK ORDER BY LOAD_DTS DESC) AS rn2
		FROM PROCURE2PAY_DEV.RAW_VAULT.SAT_MATERIAL_DOC_MKPF
	) mk ON hd.MATERIAL_DOC_HK = mk.MATERIAL_DOC_HK AND mk.rn2 = 1
	WHERE m.PURCHASE_ORDER_NUMBER IS NOT NULL AND m.PO_ITEM_NUMBER IS NOT NULL AND mk.POSTING_DATE IS NOT NULL
),
joined AS (
	SELECT
		g.yr,
		g.mo,
		CASE WHEN s.del_date IS NULL OR g.POSTING_DATE <= s.del_date THEN 1 ELSE 0 END AS on_time,
		1 AS tot
	FROM gr g
	LEFT JOIN sched s ON s.PURCHASE_ORDER_NUMBER = g.PURCHASE_ORDER_NUMBER AND s.PO_ITEM_NUMBER = g.PO_ITEM_NUMBER
	WHERE g.rn = 1
),
agg AS (
	SELECT
		yr,
		mo,
		SUM(on_time) AS on_time_cnt,
		SUM(tot) AS total_cnt
	FROM joined
	GROUP BY yr, mo
)
SELECT
	yr AS YEAR,
	mo AS MONTH,
	on_time_cnt AS ON_TIME_DELIVERIES,
	total_cnt AS TOTAL_DELIVERIES,
	CASE WHEN total_cnt > 0 THEN 100.0 * on_time_cnt / total_cnt ELSE 0 END AS DELIVERY_ACCURACY_PCT
FROM agg;


create or replace dynamic table PROCURE2PAY_DEV.INFORMATION_MART.WEIGHTED_DAYS_PAYABLE_OUTSTANDING_D(
	YEAR,
	MONTH,
	TOTAL_PAYABLES,
	TOTAL_COGS,
	WEIGHTED_DPO
) target_lag = '90 days' refresh_mode = AUTO initialize = ON_CREATE warehouse = CORTEX_ANALYST_DEV_WH
 as
WITH inv AS (
	SELECT
		s.POSTING_DATE,
		s.INV_AMOUNT_LOCAL_CURRENCY,
		ROW_NUMBER() OVER (PARTITION BY h.INVOICE_HK ORDER BY s.LOAD_DTS DESC) AS rn
	FROM PROCURE2PAY_DEV.RAW_VAULT.HUB_INVOICE h
	INNER JOIN PROCURE2PAY_DEV.RAW_VAULT.SAT_INVOICE_RBKP s ON h.INVOICE_HK = s.INVOICE_HK
	WHERE s.POSTING_DATE IS NOT NULL AND s.INV_AMOUNT_LOCAL_CURRENCY IS NOT NULL
),
by_period AS (
	SELECT
		YEAR(POSTING_DATE) AS yr,
		MONTH(POSTING_DATE) AS mo,
		INV_AMOUNT_LOCAL_CURRENCY,
		INV_AMOUNT_LOCAL_CURRENCY * 0.70 AS cogs_est
	FROM inv
	WHERE rn = 1
),
agg AS (
	SELECT
		yr,
		mo,
		SUM(INV_AMOUNT_LOCAL_CURRENCY) AS tot_payables,
		SUM(cogs_est) AS tot_cogs
	FROM by_period
	GROUP BY yr, mo
)
SELECT
	yr AS YEAR,
	mo AS MONTH,
	tot_payables AS TOTAL_PAYABLES,
	tot_cogs AS TOTAL_COGS,
	CASE WHEN tot_cogs > 0 THEN (tot_payables / tot_cogs) * 30 ELSE NULL END AS WEIGHTED_DPO
FROM agg;
