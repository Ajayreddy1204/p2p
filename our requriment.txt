# --------------------------------------------------------------
# Helper: get all bronze tables and their schemas
# --------------------------------------------------------------
def get_bronze_tables(database_name):
    tables = []
    next_token = None
    while True:
        if next_token:
            response = glue_client.get_tables(DatabaseName=database_name, NextToken=next_token)
        else:
            response = glue_client.get_tables(DatabaseName=database_name)
        for table in response['TableList']:
            table_name = table['Name']
            table_detail = glue_client.get_table(DatabaseName=database_name, Name=table_name)
            columns = []
            for col_def in table_detail['Table']['StorageDescriptor']['Columns']:
                columns.append({"name": col_def['Name'], "type": col_def['Type']})
            tables.append({"name": table_name, "columns": columns})
        next_token = response.get('NextToken')
        if not next_token:
            break
    logger.info("Found %d tables in database %s: %s", len(tables), database_name, [t['name'] for t in tables])
    return tables

def get_model_from_llm(tables_info, actual_table_names):
    prompt = f"""
You are a Data Vault 2.0 expert. Given the following SAP source tables and their columns, 
propose HUB, LINK, and SATELLITE tables.  
**CRITICAL**: Only use the source tables listed below. Do not invent new table names.

Actual source tables (exact names): {actual_table_names}

Naming conventions:
- HUB tables: HUB_<business concept> (e.g., HUB_VENDOR, HUB_PURCHASE_ORDER, HUB_COMPANY_CODE)
- LINK tables: LINK_<concept1>_<concept2> (e.g., LINK_PURCHASE_ORDER_VENDOR, LINK_VENDOR_COMPANY_CODE)
- SAT tables: SAT_<description>_<source_table> (e.g., SAT_VENDOR_LFA1, SAT_PURCHASE_ORDER_EKKO)

Rules:
- A HUB must have one business key column from a source table.
- A LINK must connect at least two HUBs via foreign key columns that exist in the source table.
- A SATELLITE must contain descriptive attributes from a single source table and link to one HUB.

Return **only** valid JSON with this structure:
{{
  "hubs": [
    {{ "hub_name": "HUB_VENDOR", "business_key_column": "lifnr", "source_tables": ["LFA1", "LFB1"] }},
    {{ "hub_name": "HUB_PURCHASE_ORDER", "business_key_column": "ebeln", "source_tables": ["EKKO"] }},
    {{ "hub_name": "HUB_COMPANY_CODE", "business_key_column": "bukrs", "source_tables": ["T001"] }}
  ],
  "links": [
    {{ "link_name": "LINK_PURCHASE_ORDER_VENDOR", 
       "hub_references": [
         {{"hub_name": "HUB_PURCHASE_ORDER", "hub_reference_column": "ebeln"}},
         {{"hub_name": "HUB_VENDOR", "hub_reference_column": "lifnr"}}
       ], 
       "source_tables": ["EKKO"] }}
  ],
  "satellites": [
    {{ "sat_name": "SAT_VENDOR_LFA1", "hub_reference": "HUB_VENDOR", 
       "attributes": ["name1", "ort01", "land1"], "source_table": "LFA1" }},
    {{ "sat_name": "SAT_PURCHASE_ORDER_EKKO", "hub_reference": "HUB_PURCHASE_ORDER",
       "attributes": ["bsart", "ekorg", "werks"], "source_table": "EKKO" }}
  ]
}}

Now, analyze the following source tables and columns, and generate the JSON:

{json.dumps(tables_info, indent=2)}

Think step by step, then output the JSON.
"""
    body = json.dumps({
        "messages": [{"role": "user", "content": [{"text": prompt}]}],
        "inferenceConfig": {"maxTokens": 4096, "temperature": 0.1, "topP": 0.9}
    })
    try:
        response = bedrock_client.invoke_model(
            modelId="amazon.nova-lite-v1:0",
            contentType="application/json",
            accept="application/json",
            body=body
        )
        response_body = json.loads(response['body'].read())
        model_output = response_body['output']['message']['content'][0]['text']
        start = model_output.find('{')
        end = model_output.rfind('}') + 1
        if start != -1 and end != 0:
            model_output = model_output[start:end]
        model_json = json.loads(model_output)
        logger.info("LLM model generated successfully: %d hubs, %d links, %d sats",
                    len(model_json.get('hubs', [])),
                    len(model_json.get('links', [])),
                    len(model_json.get('satellites', [])))
        return model_json
    except Exception as e:
        logger.error("Bedrock invocation failed: %s", e)
        raise

def write_to_s3(df, s3_path, mode="overwrite"):
    full_path = f"s3://{raw_vault_bucket}/{s3_path}"
    df.write.mode(mode).parquet(full_path)
    logger.info("Written %d rows to %s", df.count(), full_path)

# --------------------------------------------------------------
# Helper: case-insensitive column lookup
# --------------------------------------------------------------
def find_column(df, col_name):
    for c in df.columns:
        if c.lower() == col_name.lower():
            return c
    return None

# --------------------------------------------------------------
# Main
# --------------------------------------------------------------
def main():
    # 1. Load config and bronze tables
    config = read_config()
    bronze_tables = get_bronze_tables(bronze_db)
    table_map = build_table_map(bronze_tables)
    actual_table_names = list(table_map.keys())
    logger.info("Actual table names (lowercase): %s", actual_table_names)
    
    # 2. Get LLM model
    vault_model = get_model_from_llm(bronze_tables, [t['name'] for t in bronze_tables])
    
    # 3. Validate hubs (case-insensitive source table matching)
    valid_hubs = []
    for hub in vault_model.get('hubs', []):
        src_tables = hub.get('source_tables', [])
        valid_src = []
        for src in src_tables:
            src_lower = src.lower()
            if src_lower in table_map:
                valid_src.append(table_map[src_lower])
            else:
                logger.warning("Hub %s references non-existent table %s", hub['hub_name'], src)
        if valid_src:
            hub['source_tables'] = valid_src
            valid_hubs.append(hub)
        else:
            logger.warning("Hub %s has no valid source tables, skipping", hub['hub_name'])
    vault_model['hubs'] = valid_hubs
    
    # 4. Validate satellites
    valid_sats = []
    for sat in vault_model.get('satellites', []):
        src = sat['source_table']
        src_lower = src.lower()
        if src_lower in table_map:
            sat['source_table'] = table_map[src_lower]
            valid_sats.append(sat)
        else:
            logger.warning("Satellite %s references missing table %s, skipping", sat['sat_name'], src)
    vault_model['satellites'] = valid_sats
    
    # 5. Validate links
    valid_links = []
    for link in vault_model.get('links', []):
        src_tables = link.get('source_tables', [])
        valid_src = []
        for src in src_tables:
            src_lower = src.lower()
            if src_lower in table_map:
                valid_src.append(table_map[src_lower])
            else:
                logger.warning("Link %s references non-existent source table %s", link['link_name'], src)
        if valid_src:
            link['source_tables'] = valid_src
            valid_links.append(link)
        else:
            logger.warning("Link %s has no valid source tables, skipping", link['link_name'])
    vault_model['links'] = valid_links
    
    logger.info("After validation: %d hubs, %d links, %d sats",
                len(vault_model['hubs']), len(vault_model['links']), len(vault_model['satellites']))
    
    # 6. Create HUB tables
    for hub in vault_model['hubs']:
        hub_name = hub['hub_name']
        business_key_col = hub['business_key_column']
        source_tables = hub['source_tables']
        logger.info("Building HUB: %s (key: %s) from %s", hub_name, business_key_col, source_tables)
        
        dfs = []
        for src_tbl in source_tables:
            try:
                src_df = spark.table(f"{bronze_db}.{src_tbl}")
                actual_col = find_column(src_df, business_key_col)
                if actual_col is None:
                    logger.warning("Column %s not found in %s, skipping", business_key_col, src_tbl)
                    continue
                key_df = src_df.select(
                    col(actual_col).alias("business_key"),
                    lit(src_tbl).alias("record_source"),
                    current_timestamp().alias("load_date")
                ).distinct()
                dfs.append(key_df)
            except Exception as e:
                logger.error("Error reading %s: %s", src_tbl, e)
                continue
        
        if not dfs:
            logger.warning("No data for HUB %s, skipping", hub_name)
            continue
        
        hub_df = dfs[0]
        for df in dfs[1:]:
            hub_df = hub_df.union(df)
        
        hub_df = hub_df.withColumn("hub_hash_key", sha2(concat_ws("||", col("business_key")), 256))
        hub_df = hub_df.select("hub_hash_key", "business_key", "load_date", "record_source")
        write_to_s3(hub_df, f"{hub_path}/{hub_name}")
    
    # 7. Create SATELLITE tables
    for sat in vault_model['satellites']:
        sat_name = sat['sat_name']
        hub_ref = sat['hub_reference']
        attributes = sat.get('attributes', [])
        source_table = sat['source_table']
        logger.info("Building SATELLITE: %s linked to %s from %s", sat_name, hub_ref, source_table)
        
        try:
            src_df = spark.table(f"{bronze_db}.{source_table}")
        except Exception as e:
            logger.error("Cannot read source table %s: %s", source_table, e)
            continue
        
        hub_def = next((h for h in vault_model['hubs'] if h['hub_name'] == hub_ref), None)
        if not hub_def:
            logger.warning("No hub definition found for %s", hub_ref)
            continue
        business_key_col = hub_def['business_key_column']
        
        hub_path_full = f"s3://{raw_vault_bucket}/{hub_path}/{hub_ref}"
        try:
            # Only load needed columns from hub (exclude load_date to avoid ambiguity)
            hub_df = spark.read.parquet(hub_path_full).select("hub_hash_key", "business_key")
        except Exception as e:
            logger.error("Cannot read HUB %s: %s", hub_ref, e)
            continue
        
        join_col = find_column(src_df, business_key_col)
        if join_col is None:
            logger.warning("Business key column %s not found in source table %s", business_key_col, source_table)
            continue
        
        existing_attrs = [a for a in attributes if a in src_df.columns]
        if not existing_attrs:
            logger.warning("No valid attributes for satellite %s", sat_name)
            continue
        
        sat_df = src_df.join(hub_df, src_df[join_col] == hub_df["business_key"], "inner") \
                       .select(
                           hub_df["hub_hash_key"],
                           *[col(a).alias(a) for a in existing_attrs],
                           lit(source_table).alias("record_source"),
                           current_timestamp().alias("load_date")
                       ).distinct()
        
        if sat_df.count() == 0:
            logger.warning("No data for satellite %s after join", sat_name)
            continue
        
        attr_cols = [col(a) for a in existing_attrs]
        if attr_cols:
            concat_expr = concat_ws("||", *[c.cast("string") for c in attr_cols])
            sat_df = sat_df.withColumn("hash_diff", sha2(concat_expr, 256))
        
        write_to_s3(sat_df, f"{sat_path}/{sat_name}")
    
    # 8. Create LINK tables (fixed ambiguous load_date)
    for link in vault_model['links']:
        try:
            link_name = link['link_name']
            hub_refs = link['hub_references']
            source_tables = link['source_tables']
            logger.info("Building LINK: %s from %s", link_name, source_tables)
            
            # Load all referenced hubs (only hub_hash_key and business_key)
            hubs_dict = {}
            valid = True
            for ref in hub_refs:
                hub_nm = ref['hub_name']
                hub_path_full = f"s3://{raw_vault_bucket}/{hub_path}/{hub_nm}"
                try:
                    hub_df = spark.read.parquet(hub_path_full).select("hub_hash_key", "business_key")
                    hubs_dict[hub_nm] = hub_df
                    logger.info("Loaded HUB %s (%d rows)", hub_nm, hub_df.count())
                except Exception as e:
                    logger.error("Cannot load HUB %s: %s", hub_nm, e)
                    valid = False
                    break
            if not valid:
                continue
            
            link_dfs = []
            for src_tbl in source_tables:
                try:
                    src_df = spark.table(f"{bronze_db}.{src_tbl}")
                    # Drop any existing load_date or record_source from source to avoid ambiguity
                    if "load_date" in src_df.columns:
                        src_df = src_df.drop("load_date")
                    if "record_source" in src_df.columns:
                        src_df = src_df.drop("record_source")
                    logger.info("Processing source table %s with %d columns", src_tbl, len(src_df.columns))
                except Exception as e:
                    logger.error("Cannot read source table %s: %s", src_tbl, e)
                    continue
                
                current_df = src_df
                hub_keys = []
                ok = True
                for ref in hub_refs:
                    hub_nm = ref['hub_name']
                    hub_ref_col = ref['hub_reference_column']
                    actual_col = find_column(current_df, hub_ref_col)
                    if actual_col is None:
                        logger.warning("Column %s not found in %s for link %s", hub_ref_col, src_tbl, link_name)
                        ok = False
                        break
                    hub_df = hubs_dict[hub_nm]
                    # Join on foreign key = hub.business_key
                    current_df = current_df.join(hub_df, current_df[actual_col] == hub_df["business_key"], "inner")
                    # Rename hub_hash_key to unique name
                    current_df = current_df.withColumnRenamed("hub_hash_key", f"hk_{hub_nm}")
                    hub_keys.append(f"hk_{hub_nm}")
                if not ok:
                    continue
                
                # Generate link hash key
                hash_cols = [col(f"hk_{ref['hub_name']}") for ref in hub_refs]
                link_hash_expr = sha2(concat_ws("||", *hash_cols), 256)
                current_df = current_df.withColumn("link_hash_key", link_hash_expr) \
                                       .withColumn("load_date", current_timestamp()) \
                                       .withColumn("record_source", lit(src_tbl))
                select_cols = ["link_hash_key"] + hub_keys + ["load_date", "record_source"]
                current_df = current_df.select(*select_cols).distinct()
                logger.info("Link %s from source %s has %d rows after distinct", link_name, src_tbl, current_df.count())
                link_dfs.append(current_df)
            
            if not link_dfs:
                logger.warning("No data for LINK %s", link_name)
                continue
            
            final_link_df = link_dfs[0]
            for df in link_dfs[1:]:
                final_link_df = final_link_df.union(df)
            
            write_to_s3(final_link_df, f"{link_path}/{link_name}")
        except Exception as e:
            logger.error("Failed to create link %s: %s", link.get('link_name', 'unknown'), e, exc_info=True)
    
    logger.info("Data Vault generation completed successfully")

if __name__ == "__main__":
    main()
